    for j in range(N):
        T[i][j] = -0.5*(D[i][j]**2 - 1/N*dis_line[i] + 1/(N*N)*dis_sum)
eigVal,eigVec = np.linalg.eig(T) 
#取最大的两个特征值
eigValSorted_indices = np.argsort(eigVal)
#提取d个最大特征向量

X = np.dot(topd_eigVec, np.diag(eigVal[eigValSorted_indices[:-2-1:-1]]))
